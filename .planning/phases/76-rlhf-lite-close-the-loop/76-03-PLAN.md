```sql
-- Tracks which feedback signals contributed to which improvements
-- Used for "close the loop" notifications (REQ-L1)
CREATE TABLE IF NOT EXISTS feedback_improvements (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  improvement_type TEXT NOT NULL CHECK (improvement_type IN ('prompt_patch', 'bug_fix', 'feature')),
  title TEXT NOT NULL,
  description TEXT,
  patch_id UUID REFERENCES prompt_patches(id) ON DELETE SET NULL,
  insight_id UUID REFERENCES feedback_insights(id) ON DELETE SET NULL,
  signal_ids UUID[] NOT NULL DEFAULT '{}',  -- feedback signals that contributed
  user_ids UUID[] NOT NULL DEFAULT '{}',    -- users who gave the feedback
  severity TEXT NOT NULL DEFAULT 'medium' CHECK (severity IN ('low', 'medium', 'high', 'critical')),
  resolved_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  notified_at TIMESTAMPTZ,                 -- when digest was sent
  metadata JSONB NOT NULL DEFAULT '{}',
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_feedback_improvements_resolved ON feedback_improvements(resolved_at);
CREATE INDEX idx_feedback_improvements_notified ON feedback_improvements(notified_at) WHERE notified_at IS NULL;

-- Track user notification preferences for feedback digests
-- This extends existing notification_preferences if it exists, or creates a lightweight table
CREATE TABLE IF NOT EXISTS feedback_digest_preferences (
  user_id UUID PRIMARY KEY,
  enabled BOOLEAN NOT NULL DEFAULT false,   -- opt-in only (REQ-L4)
  last_sent_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
```

```typescript
export interface ImprovementEntry {
  id: string
  improvementType: 'prompt_patch' | 'bug_fix' | 'feature'
  title: string
  description: string | null
  patchId: string | null
  insightId: string | null
  signalIds: string[]
  userIds: string[]

  notifiedAt: string | null

}
```

Functions:

1. **`logImprovement(params: { improvementType: 'prompt_patch' | 'bug_fix' | 'feature', title: string, description?: string, patchId?: string, insightId?: string, signalIds: string[], severity?: string }): Promise<ImprovementEntry>`**
   - INSERT into feedback_improvements
   - Derive user_ids from the signal_ids: `SELECT DISTINCT user_id FROM feedback_signals WHERE id = ANY($signalIds)`
   - Return created row

2. **`logImprovementFromPatch(patchId: string): Promise<ImprovementEntry | null>`**
   - Get the patch from prompt_patches
   - If patch has source_signal_ids, use those
   - Create improvement entry with type 'prompt_patch'
   - Link back to the patch and its source insight

3. **`getImprovementsForUser(userId: string, since?: string): Promise<ImprovementEntry[]>`**
   - SELECT from feedback_improvements WHERE $userId = ANY(user_ids)
   - Filter by resolved_at > since (default 30 days ago) — REQ-L3 staleness cutoff
   - Filter by severity >= 'medium' — REQ-L3 severity threshold (skip 'low')
   - ORDER BY resolved_at DESC

4. **`getRecentImprovements(since?: string): Promise<ImprovementEntry[]>`**
   - SELECT from feedback_improvements WHERE resolved_at > since
   - Default since = 30 days ago
   - Filter severity >= 'medium'

5. **`markAsNotified(improvementIds: string[]): Promise<void>`**
   - UPDATE feedback_improvements SET notified_at = NOW() WHERE id = ANY($ids)

**app/api/cron/feedback-digest/route.ts** — Cron-triggered monthly digest.

Follow existing cron route patterns in the project.

```typescript
import { NextRequest, NextResponse } from "next/server"
import { sendFeedbackDigest } from "@/lib/rlhf/close-the-loop-digest"
import { logger } from "@/lib/logger"

/**
 * GET /api/cron/feedback-digest
 * Monthly cron job to send "feedback improvements" digest emails.
 * Protected by CRON_SECRET header.
 */
export async function GET(request: NextRequest) {
  // Verify cron secret
  const cronSecret = request.headers.get("authorization")?.replace("Bearer ", "")
  if (cronSecret !== process.env.CRON_SECRET) {
    return NextResponse.json({ error: "Unauthorized" }, { status: 401 })
  }

  try {
    logger.log("[Cron] Starting feedback digest send")
    const result = await sendFeedbackDigest()
    logger.log(`[Cron] Feedback digest complete: ${result.sent} sent, ${result.skipped} skipped, ${result.errors} errors`)

    return NextResponse.json({
      success: true,
      ...result,
    })
  } catch (error) {
    console.error("[Cron] Feedback digest failed:", error)
    return NextResponse.json(
      { success: false, error: "Digest send failed" },
      { status: 500 }
    )
  }
}
```
