```typescript
/**
 * Prompt Patch Generator
 *
 * Phase 76 REQ-R2: When pattern detection identifies a recurring complaint,
 * generate a supplemental prompt patch for that coaching topic.
 *
 * Uses LLM to convert feedback patterns into actionable prompt instructions
 * that are appended to FRED's mutable supplemental layer.
 */
import { generateText } from "ai"
import { getModel } from "@/lib/ai/providers"
import { getModelForTier } from "@/lib/ai/tier-routing"
import { insertPromptPatch } from "@/lib/db/prompt-patches"
import { getFewShotExamplesByTopic } from "@/lib/feedback/few-shot-examples"
import type { FeedbackInsight, PromptPatchInsert } from "@/lib/feedback/types"

/**
 * Generate a prompt patch from a feedback insight (cluster).
 *
 * Takes the insight's theme, description, category, and related few-shot examples,
 * then uses an LLM to draft a supplemental prompt instruction.
 *
 * The generated patch is stored with status 'pending' awaiting admin approval.
 */
export async function generatePatchFromInsight(
  insight: FeedbackInsight
): Promise<{ patchId: string; content: string } | null> {
  try {
    const topic = insight.category || "general"

    // Get few-shot examples for this topic to provide context
    const examples = await getFewShotExamplesByTopic(topic, 5)

    const positiveExamples = examples.positive
      .slice(0, 3)
      .map((e, i) => `  ${i + 1}. "${e.message_content.slice(0, 200)}"`)
      .join("\n")

    const negativeExamples = examples.negative
      .slice(0, 3)
      .map((e, i) => `  ${i + 1}. "${e.message_content.slice(0, 200)}" (Category: ${e.category || "unknown"})`)
      .join("\n")

    const prompt = `You are helping improve an AI startup mentor called FRED (on joinsahara.com).

FRED has received recurring negative feedback about this topic:
- Theme: ${insight.title}
- Description: ${insight.description || "No description"}
- Category: ${insight.category || "mixed"}
- Signal count: ${insight.signal_count} complaints

${positiveExamples ? `Examples of responses users LIKED on this topic:\n${positiveExamples}\n` : ""}
${negativeExamples ? `Examples of responses users DISLIKED on this topic:\n${negativeExamples}\n` : ""}

Write a SHORT supplemental prompt instruction (3-5 sentences max) that:
1. Addresses the specific complaint pattern
2. Gives FRED clear guidance on how to handle this topic better
3. Does NOT contradict FRED's core identity (blunt truth-teller, startup mentor, reframe-before-prescribe)
4. Uses imperative voice ("When X, do Y" or "For X topics, always Z")

Return ONLY the instruction text, no explanations or metadata.`

    const providerKey = getModelForTier("pro", "chat")
    const model = getModel(providerKey)

    const result = await generateText({
      model,
      prompt,
      maxTokens: 300,
      temperature: 0.3,
    })

    const patchContent = result.text.trim()

    if (!patchContent || patchContent.length < 20) {
      console.warn("[patch-generator] Generated patch too short, skipping")
      return null
    }

    // Store as pending patch
    const patch = await insertPromptPatch({
      title: `[Auto] ${insight.title}`,
      content: patchContent,
      category: insight.category,
      source: "feedback",
      source_signal_ids: insight.signal_ids,
      source_insight_id: insight.id,
      version: 1,
      status: "pending",
      ab_experiment_id: null,
      deployed_at: null,
      graduated_at: null,
      rejected_at: null,
      rejection_reason: null,
      performance_metrics: {},
      previous_versions: [],
      created_by: "system",
      metadata: {
        generated_at: new Date().toISOString(),
        insight_severity: insight.severity,
        signal_count: insight.signal_count,
        few_shot_positive_count: examples.positive.length,
        few_shot_negative_count: examples.negative.length,
      },
    })

    return { patchId: patch.id, content: patchContent }
  } catch (err) {
    console.error("[patch-generator] Failed to generate patch:", err)
    return null
  }
}

/**
 * Generate a prompt patch from raw text (manual creation via admin).
 * Stored immediately with status 'draft'.
 */
export async function generatePromptPatch(params: {
  title: string
  content: string
  category?: string
  createdBy: string
}): Promise<string> {
  const patch = await insertPromptPatch({
    title: params.title,
    content: params.content,
    category: params.category || null,
    source: "manual",
    source_signal_ids: [],
    source_insight_id: null,
    version: 1,
    status: "draft",
    ab_experiment_id: null,
    deployed_at: null,
    graduated_at: null,
    rejected_at: null,
    rejection_reason: null,
    performance_metrics: {},
    previous_versions: [],
    created_by: params.createdBy,
    metadata: { created_at: new Date().toISOString() },
  })

  return patch.id
}
```

```typescript
import { NextRequest, NextResponse } from "next/server"
import { createServiceClient } from "@/lib/supabase/server"

export async function GET(req: NextRequest) {
  try {
    const supabase = createServiceClient()
    const { searchParams } = new URL(req.url)
    const status = searchParams.get("status") || "pending"
    const limit = Math.min(parseInt(searchParams.get("limit") || "50"), 100)

    let query = supabase
      .from("prompt_patches")
      .select("*")
      .order("created_at", { ascending: false })
      .limit(limit)

    if (status !== "all") {
      query = query.eq("status", status)
    }

    const { data, error } = await query
    if (error) throw error

    return NextResponse.json({ patches: data ?? [], total: data?.length ?? 0 })
  } catch (err) {
    console.error("[admin/patches] GET error:", err)
    return NextResponse.json({ error: "Failed to fetch patches" }, { status: 500 })
  }
}
```

```typescript
import { NextRequest, NextResponse } from "next/server"
import { updatePatchStatus } from "@/lib/db/prompt-patches"

export async function PATCH(
  req: NextRequest,
  { params }: { params: Promise<{ id: string }> }
) {
  try {
    const { id } = await params
    const body = await req.json()
    const { status, rejection_reason } = body

    const validStatuses = ["draft", "pending", "approved", "deployed", "rejected", "graduated"]
    if (!validStatuses.includes(status)) {
      return NextResponse.json({ error: "Invalid status" }, { status: 400 })
    }

    const extra: Record<string, unknown> = {}
    if (status === "rejected") {
      extra.rejection_reason = rejection_reason || null
      extra.rejected_at = new Date().toISOString()
    }
    if (status === "deployed") {
      extra.deployed_at = new Date().toISOString()
    }
    if (status === "graduated") {
      extra.graduated_at = new Date().toISOString()
    }

    const patch = await updatePatchStatus(id, status, extra as Record<string, string>)
    return NextResponse.json({ patch })
  } catch (err) {
    console.error("[admin/patches] PATCH error:", err)
    return NextResponse.json({ error: "Failed to update patch" }, { status: 500 })
  }
}
```

**app/api/admin/feedback/patches/[id]/experiment/route.ts:**
```typescript
import { NextRequest, NextResponse } from "next/server"
import { getPatchById, updatePatchStatus } from "@/lib/db/prompt-patches"
import { createExperiment } from "@/lib/ai/ab-testing"

export async function POST(
  req: NextRequest,
  { params }: { params: Promise<{ id: string }> }
) {
  try {
    const { id } = await params
    const patch = await getPatchById(id)

    if (!patch) {
      return NextResponse.json({ error: "Patch not found" }, { status: 404 })
    }

    if (patch.status !== "approved" && patch.status !== "pending") {
      return NextResponse.json(
        { error: `Patch status must be 'approved' or 'pending', got '${patch.status}'` },
        { status: 400 }
      )
    }

    // Create A/B experiment: control (no patch) vs treatment (with patch)
    const experimentName = `patch-${patch.id.slice(0, 8)}-${patch.category || "general"}`
    const experimentId = await createExperiment(
      experimentName,
      `A/B test for prompt patch: ${patch.title}`,
      [
        { variantName: "control", trafficPercentage: 50 },
        {
          variantName: "treatment",
          trafficPercentage: 50,
          configOverrides: {
            patchId: patch.id,
            patchContent: patch.content,
          },
        },
      ],
      "system"
    )

    // Update patch status to deployed with experiment link
    await updatePatchStatus(id, "deployed", {
      ab_experiment_id: experimentName,
      deployed_at: new Date().toISOString(),
    })

    return NextResponse.json({
      experimentId,
      experimentName,
      message: "A/B test created. Control (50%) vs treatment with patch (50%).",
    })
  } catch (err) {
    console.error("[admin/patches/experiment] POST error:", err)
    return NextResponse.json({ error: "Failed to create experiment" }, { status: 500 })
  }
}
```
