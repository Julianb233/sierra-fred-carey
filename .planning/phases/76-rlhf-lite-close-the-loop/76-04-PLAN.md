/**
 * Improvements Digest
 *
 * Phase 76 REQ-L2/L3/L4: Monthly digest email showing founders
 * what FRED improved based on their feedback.
 *
 * Filters: 30-day staleness cutoff, medium+ severity only,
 * notification preferences respected.
 */
import { createServiceClient } from "@/lib/supabase/server"

const STALENESS_CUTOFF_DAYS = 30
const MIN_SEVERITY_ORDER: Record<string, number> = {
  low: 1,
  medium: 2,
  high: 3,
  critical: 4,
}
const MIN_SEVERITY_THRESHOLD = 2 // medium

export interface DigestImprovement {
  patchTitle: string
  category: string
  signalCount: number
  improvementDelta: number | null
  deployedAt: string
}

export interface DigestData {
  improvements: DigestImprovement[]
  totalSignalsContributed: number
  periodStart: string
  periodEnd: string
}

/**
 * Gather data for the improvements digest.
 * Returns improvements from the last 30 days with severity >= medium.
 */
export async function gatherDigestData(): Promise<DigestData> {
  const supabase = createServiceClient()

  const periodEnd = new Date()
  const periodStart = new Date()
  periodStart.setDate(periodStart.getDate() - STALENESS_CUTOFF_DAYS)

  // Get graduated or deployed patches from the last 30 days
  const { data: patches, error } = await supabase
    .from("prompt_patches")
    .select("*, feedback_insights!prompt_patches_source_insight_id_fkey(severity)")
    .in("status", ["deployed", "graduated"])
    .gte("deployed_at", periodStart.toISOString())
    .order("deployed_at", { ascending: false })

  if (error || !patches) {
    console.error("[digest] Failed to fetch patches:", error?.message)
    return {
      improvements: [],
      totalSignalsContributed: 0,
      periodStart: periodStart.toISOString(),
      periodEnd: periodEnd.toISOString(),
    }
  }

  // Filter by severity threshold
  const improvements: DigestImprovement[] = []
  let totalSignals = 0

  for (const patch of patches) {
    // Check severity from source insight
    const insight = Array.isArray(patch.feedback_insights)
      ? patch.feedback_insights[0]
      : patch.feedback_insights
    const severity = insight?.severity || "medium"
    const severityOrder = MIN_SEVERITY_ORDER[severity] || 0

    if (severityOrder < MIN_SEVERITY_THRESHOLD) continue

    const metrics = (patch.performance_metrics as Record<string, unknown>) || {}
    const signalCount = (patch.source_signal_ids as string[])?.length || 0
    totalSignals += signalCount

    improvements.push({
      patchTitle: (patch.title as string).replace(/^\[Auto\]\s*/, ""),
      category: (patch.category as string) || "General",
      signalCount,
      improvementDelta: (metrics.improvement_delta as number) || null,
      deployedAt: patch.deployed_at as string,
    })
  }

  return {
    improvements,
    totalSignalsContributed: totalSignals,
    periodStart: periodStart.toISOString(),
    periodEnd: periodEnd.toISOString(),
  }
}

/**
 * Get list of user IDs eligible for the digest.
 * Eligible: users who have given feedback AND have not opted out.
 */
export async function getDigestRecipients(): Promise<string[]> {
  const supabase = createServiceClient()

  // Get unique users who have submitted feedback with consent
  const { data, error } = await supabase
    .from("feedback_signals")
    .select("user_id")
    .eq("consent_given", true)

  if (error || !data) return []

  // Deduplicate
  const uniqueUsers = [...new Set(data.map((d) => d.user_id as string))]
  return uniqueUsers
}

/**
 * Send the improvements digest to a single user via Resend.
 * Returns true if sent successfully.
 */
export async function sendDigestToUser(
  userId: string,
  userEmail: string,
  digestData: DigestData
): Promise<boolean> {
  try {
    // Check notification preferences
    // Use dynamic import to avoid circular deps
    const resendApiKey = process.env.RESEND_API_KEY
    if (!resendApiKey) {
      console.warn("[digest] RESEND_API_KEY not set, skipping email")
      return false
    }

    if (digestData.improvements.length === 0) {
      return false // Nothing to report
    }

    // Build plain text version for simplicity
    const improvementsList = digestData.improvements
      .map((imp) => {
        const delta = imp.improvementDelta
          ? ` (${imp.improvementDelta > 0 ? "+" : ""}${(imp.improvementDelta * 100).toFixed(0)}% improvement)`
          : ""
        return `- ${imp.patchTitle}${delta}`
      })
      .join("\n")

    const response = await fetch("https://api.resend.com/emails", {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        Authorization: `Bearer ${resendApiKey}`,
      },
      body: JSON.stringify({
        from: "FRED <notifications@joinsahara.com>",
        to: userEmail,
        subject: "What we improved from your feedback",
        html: `
          <div style="font-family: sans-serif; max-width: 600px; margin: 0 auto; padding: 20px;">
            <div style="text-align: center; margin-bottom: 30px;">
              <h1 style="color: #ff6a1a; font-size: 24px; margin: 0;">Sahara</h1>
              <p style="color: #666; font-size: 14px;">Improvements from your feedback</p>
            </div>

            <p style="font-size: 16px; color: #333;">
              Your feedback made FRED better. Here's what changed in the last 30 days:
            </p>

            ${digestData.improvements
              .map(
                (imp) => `
              <div style="border-left: 3px solid #ff6a1a; padding: 12px 16px; margin: 16px 0; background: #fafafa; border-radius: 0 8px 8px 0;">
                <strong style="color: #333;">${imp.patchTitle}</strong>
                <br/>
                <span style="color: #666; font-size: 14px;">
                  Category: ${imp.category}
                  ${imp.improvementDelta ? ` | ${imp.improvementDelta > 0 ? "+" : ""}${(imp.improvementDelta * 100).toFixed(0)}% thumbs-up improvement` : ""}
                  | Based on ${imp.signalCount} feedback signal${imp.signalCount !== 1 ? "s" : ""}
                </span>
              </div>
            `
              )
              .join("")}

            <p style="font-size: 14px; color: #666; margin-top: 30px;">
              Total feedback signals that drove these improvements: ${digestData.totalSignalsContributed}
            </p>

            <div style="text-align: center; margin-top: 30px;">
              <a href="https://joinsahara.com/chat" style="background: #ff6a1a; color: white; padding: 12px 24px; border-radius: 8px; text-decoration: none; font-weight: bold;">
                Continue with FRED
              </a>
            </div>

            <p style="font-size: 12px; color: #999; margin-top: 40px; text-align: center;">
              You're receiving this because you submitted feedback on Sahara.
              <br/>
              <a href="https://joinsahara.com/settings/notifications" style="color: #999;">Manage notification preferences</a>
            </p>
          </div>
        `,
        text: `Improvements from your feedback\n\nYour feedback made FRED better. Here's what changed:\n\n${improvementsList}\n\nTotal feedback signals: ${digestData.totalSignalsContributed}\n\nContinue with FRED: https://joinsahara.com/chat`,
      }),
    })

    return response.ok
  } catch (err) {
    console.error(`[digest] Failed to send to ${userId}:`, err)
    return false
  }
}

import { NextRequest, NextResponse } from "next/server"
import { createServiceClient } from "@/lib/supabase/server"

export async function POST(req: NextRequest) {
  try {
    const { gatherDigestData, getDigestRecipients, sendDigestToUser } =
      await import("@/lib/feedback/improvements-digest")

    // Gather improvement data
    const digestData = await gatherDigestData()

    if (digestData.improvements.length === 0) {
      return NextResponse.json({
        sent: 0,
        message: "No improvements to report in the last 30 days",
      })
    }

    // Get recipients
    const recipientIds = await getDigestRecipients()

    if (recipientIds.length === 0) {
      return NextResponse.json({
        sent: 0,
        message: "No eligible recipients found",
      })
    }

    // Fetch user emails
    const supabase = createServiceClient()
    const { data: users, error } = await supabase
      .from("profiles")
      .select("id, email")
      .in("id", recipientIds.slice(0, 500)) // Cap at 500 per batch

    if (error || !users) {
      return NextResponse.json(
        { error: "Failed to fetch user emails" },
        { status: 500 }
      )
    }

    // Send to each user
    let sent = 0
    let failed = 0

    for (const user of users) {
      if (!user.email) continue

      const success = await sendDigestToUser(
        user.id as string,
        user.email as string,
        digestData
      )

      if (success) sent++
      else failed++
    }

    return NextResponse.json({
      sent,
      failed,
      improvements: digestData.improvements.length,
      recipients: users.length,
      message: `Digest sent to ${sent} users (${failed} failed)`,
    })
  } catch (err) {
    console.error("[admin/digest] POST error:", err)
    return NextResponse.json(
      { error: "Failed to send digest" },
      { status: 500 }
    )
  }
}
