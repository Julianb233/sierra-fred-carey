```typescript
import { RETENTION_DAYS } from './constants'

/**
 * Calculate expiry date for feedback data (90-day GDPR retention).
 */
export function calculateExpiryDate(fromDate: Date = new Date()): string {
  const expiry = new Date(fromDate)
  expiry.setDate(expiry.getDate() + RETENTION_DAYS)
  return expiry.toISOString()
}

/**
 * Validate that consent was given before storing feedback.
 * Returns the signal data with consent fields populated.
 */
export function applyConsent<T extends { consent_given?: boolean; expires_at?: string | null }>(
  signal: T,
  hasConsent: boolean
): T {
  if (!hasConsent) {
    throw new Error('Feedback cannot be stored without explicit user consent')
  }
  return {
    ...signal,
    consent_given: true,
    expires_at: calculateExpiryDate(),
  }
}

/**
 * Minimize feedback data before AI analysis.
 * Strips PII and keeps only the analytical fields.
 */
export function minimizeForAnalysis(signal: {
  signal_type: string
  rating: number | null
  category: string | null
  comment: string | null
  sentiment_score: number | null
  channel: string
  user_tier: string
}): {
  signal_type: string
  rating: number | null
  category: string | null
  comment_length: number  // length only, not content
  sentiment_score: number | null
  channel: string
  user_tier: string
} {
  return {
    signal_type: signal.signal_type,
    rating: signal.rating,
    category: signal.category,
    comment_length: signal.comment?.length ?? 0,
    sentiment_score: signal.sentiment_score,
    channel: signal.channel,
    user_tier: signal.user_tier,
  }
}

/**
 * Check if user has given feedback consent.
 * Reads from user's profile metadata.
 */
export async function getUserConsentStatus(
  supabase: ReturnType<typeof import('@/lib/supabase/server').createServiceClient>,
  userId: string
): Promise<boolean> {
  const { data } = await supabase
    .from('profiles')
    .select('metadata')
    .eq('id', userId)
    .single()
  return data?.metadata?.feedback_consent === true
}

/**
 * Set user's feedback consent status.
 */
export async function setUserConsent(
  supabase: ReturnType<typeof import('@/lib/supabase/server').createServiceClient>,
  userId: string,
  consent: boolean
): Promise<void> {
  const { data: existing } = await supabase
    .from('profiles')
    .select('metadata')
    .eq('id', userId)
    .single()
  const metadata = { ...(existing?.metadata || {}), feedback_consent: consent, feedback_consent_at: consent ? new Date().toISOString() : null }
  const { error } = await supabase
    .from('profiles')
    .update({ metadata })
    .eq('id', userId)
  if (error) throw error
}
```

### Step 3: Document lint/test baseline and create CI delta script

**Files:**
- `scripts/ci/baseline-snapshot.json` (new)
- `scripts/ci/check-delta.sh` (new)

**`scripts/ci/baseline-snapshot.json`:**
Run the current lint and test suite, capture the baseline:
```bash
# Capture current state
npx eslint . --format json 2>/dev/null | node -e "const d=require('fs').readFileSync('/dev/stdin','utf8');const j=JSON.parse(d);console.log(JSON.stringify({lint_errors:j.reduce((a,f)=>a+f.errorCount,0),lint_warnings:j.reduce((a,f)=>a+f.warningCount,0)}))"

npm run test -- --reporter=json 2>/dev/null | node -e "..."
```

Create a JSON file:
```json
{
  "captured_at": "2026-03-06",
  "lint": {
    "errors": <actual count>,
    "warnings": <actual count>,
    "note": "Pre-existing. Do not increase. Tracked for delta checking."
  },
  "tests": {
    "total": <actual count>,
    "passing": <actual count>,
    "failing": <actual count>,
    "note": "Pre-existing failures in profile-creation and get-started flows."
  }
}
```

**`scripts/ci/check-delta.sh`:**
A bash script that:
1. Runs lint, counts errors/warnings
2. Runs tests, counts pass/fail
3. Compares against `baseline-snapshot.json`
4. Fails (exit 1) if lint errors increased OR test failures increased
5. Succeeds (exit 0) if counts are same or improved
6. Prints clear output: `LINT: 335 errors (baseline: 335) -- OK` or `LINT: 340 errors (baseline: 335) -- FAIL (+5 new errors)`

The script should be executable (`chmod +x`) and usable in CI:
```bash
#!/bin/bash
# Usage: scripts/ci/check-delta.sh
# Exit 0 if no regressions, exit 1 if new errors introduced
```
