```typescript
/**
 * Increment the redirect count for a specific stage gate intent key.
 * Key format: "{oasesStage}:{intentCategory}" e.g., "build:pitch_deck_creation"
 */
export async function incrementStageGateRedirect(
  userId: string,
  intentKey: string
): Promise<number> {
  const supabase = createServiceClient();
  // Read current redirects
  const { data } = await supabase
    .from("fred_conversation_state")
    .select("stage_gate_redirects")
    .eq("user_id", userId)
    .single();

  const current = (data?.stage_gate_redirects as Record<string, number>) || {};
  const newCount = (current[intentKey] || 0) + 1;
  current[intentKey] = newCount;

  await supabase
    .from("fred_conversation_state")
    .update({ stage_gate_redirects: current })
    .eq("user_id", userId);

  return newCount;
}

/**
 * Get all stage gate redirect counts for a user.
 */
export async function getStageGateRedirects(
  userId: string
): Promise<Record<string, number>> {
  const supabase = createServiceClient();
  const { data } = await supabase
    .from("fred_conversation_state")
    .select("stage_gate_redirects")
    .eq("user_id", userId)
    .single();

  return (data?.stage_gate_redirects as Record<string, number>) || {};
}
```

```typescript
// Phase 80: Stage Gate Enforcement
let stageGateBlock = "";
let stageGateProactiveBlock = "";

if (conversationState) {
  // Determine current Oases stage label for proactive guidance
  const currentStepIdx = STEP_ORDER.indexOf(conversationState.currentStep);
  let currentOasesStage = "Clarity";
  for (const [, config] of Object.entries(OASES_STAGE_CONFIG)) {
    const minIdx = STEP_ORDER.indexOf(config.minStep);
    if (currentStepIdx >= minIdx) {
      currentOasesStage = config.label;
    }
  }

  // Build proactive guidance block (always injected)
  stageGateProactiveBlock = buildProactiveGuidanceBlock(
    conversationState.currentStep,
    currentOasesStage
  );

  // Run stage gate validation on the user's message
  const validator = new StageGateValidator(
    conversationState.currentStep,
    conversationState.stepStatuses,
    conversationState.stageGateRedirects || {}
  );
  const gateResult = validator.validate(message);

  if (!gateResult.allowed) {
    stageGateBlock = gateResult.redirectBlock || "";

    // Persist the redirect count (fire-and-forget)
    const intentKey = `${gateResult.intentStage}:${gateResult.detectedIntent}`;
    incrementStageGateRedirect(userId, intentKey).catch((err) => {
      console.warn("[FRED Chat] Failed to increment stage gate redirect (non-blocking):", err);
    });
  } else if (gateResult.isOverride) {
    // Log override for analytics
    serverTrack(userId, "stage_gate_override", {
      intent: gateResult.detectedIntent,
      intentStage: gateResult.intentStage,
      currentStep: gateResult.currentStep,
    });
  }
}
```
